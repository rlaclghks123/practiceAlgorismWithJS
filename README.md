#### 약수

- 순서쌍의 개수 참조

```JS
// 처음에 1부터 약수를 구하고 싶은 숫자를 temp값으로 나눠준다.
let temp = 1;

//약수를 구하고 싶은 숫자 n에 대해 temp값이 n보다 작거나 같은경우 while문을 통해 반복
while (temp <= n) {

    //만약 나누어 떨어진다면 약수이므로, arr에 담아준다.
  if (n % temp === 0) {
    arr.push(temp);
  }

  //약수를 구하고 난뒤, temp값을 +1 해줌으로서 점차 증가하여 모든 수의 약수를 구해준다.
  temp++;
}
```
